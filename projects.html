Caesar Cipher Encode/Decode

                        
                            import sys, os
                            import getopt
                            import time
                            
                            def num2chr(n):
                                assert 0 <= n <= 25
                                return chr(ord('a') + n)
                            
                            def chr2num(c):
                                assert 'a' <= c <= 'z'
                                return ord(c) - ord('a')
                            
                            def rot(x, n):
                                return num2chr((chr2num(x) + n) % 26)
                            
                            def decrypt(text, n):
                                z = 26 - n
                                code = ''.join([ rot(c, z) for c in text ])
                                return code
                            
                            def encrypt(text, n):
                                code = ''.join([ rot(c, n) for c in text ])
                                return code
                            
                            if __name__ == '__main__':
                                import os, sys
                                import getopt   
                            
                                n = 13
                                oCrypt = 'e'
                                opts, args = getopt.getopt(sys.argv[1:], 'hedn:')
                                for opt, arg in opts:
                                    if opt == '-h':
                                        print(f'Usage: {sys.argv[0]} [-h] [-e|d] [ -n rot ] ')
                                        sys.exit()
                                    elif opt == '-n': n = int(arg)
                                    elif opt == '-e': oCrypt = 'e'
                                    elif opt == '-d': oCrypt = 'd'
                            
                                if oCrypt == 'e':
                                    plain = ''.join(args).lower()
                                    code = encrypt(plain, n)
                                    print(code)
                                if oCrypt == 'd':
                                    code = ''.join(args).lower()
                                    plain = decrypt(code, n)
